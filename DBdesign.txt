Accounts(username char(20), email char(40), password char(45), level int, PRIMARY KEY (username));
Customers(ssn char(14), username char(20), lastname, firstname, address, city, state, zipcode, phone, credit_card_num, PRIMARY KEY (ssn), FOREIGN KEY(username) REFERENCES Accounts);
Managers(ssn char(14), username char(20), PRIMARY KEY(ssn), FOREIGN KEY (username) REFERENCES Accounts);
